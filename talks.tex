
\resheading{Talks/Workshops}
\begin{itemize}

\item
    \ressubheading{The Trace Problem and Fractional Order Differentiation}{Fall 2007}{Boulder, CO}{}
    \begin{itemize}
        \resitem{Discuss what happens when one takes elements of Sobolev Spaces and restricts them to lower dimensional subsets.}
    \end{itemize}

\item
    \ressubheading{HTML Workshop}{Fall 2006}{Boulder, CO}{}
    \begin{itemize}
        \resitem{Workshop for first year graduate students on how to create course web pages on the department server.  Brief introduction 
                    to working in a Unix environment.}
    \end{itemize}

\item
    \ressubheading{C++/Java Workshop}{Fall 2006}{Boulder, CO}{}
    \begin{itemize}
        \resitem{Gave an introduction to using the C/C++/Java tools available to graduate students in the Department of Mathematics.}
    \end{itemize}

\item
	\ressubheading{Distributions and the Direct Delta Function}{Fall 2004}{Boulder, CO}{}
	\begin{itemize}
		\resitem{It is often stated that ``the direct delta function isn't a function''...  So what exactly is it?  
                This talk gave an overview of distribution theory and weak derivatives.}
	\end{itemize}

\item
	\ressubheading{Sobolev Spaces, Orlicz Spaces and Embedding Theorems}{Spring 2003}{Boulder, CO}{(M.A. Presentation)}
	\begin{itemize}
		\resitem{Examine the so-called critical case $mp=n$ for target spaces to embed the sobolev space $W_{m,p}(\mathbb{R}^n)$.  Show
                    how one can not embed into $L^{\infty}$, but a suitable alternative can be found in the Orlicz Spaces.}
	\end{itemize}

\item
	\ressubheading{Approximation in Hilbert Spaces, II}{Fall 2002}{Boulder, CO}{}
	\begin{itemize}
		\resitem{Part two of a two part talk about using least squares techniques in Hilbert Spaces.  Show how one can obtain uniform estimates
                    for approximations of functions using Sobolev norms in the $W^{k,2}$ spaces.}
	\end{itemize}

\item
	\ressubheading{Approximation in Hilbert Spaces, I}{Fall 2002}{Boulder, CO}{}
	\begin{itemize}
		\resitem{Part one of a two part talk about using least squares techniques in Hilbert Spaces.  Introduce least squares, 
                    derive the error minimizing property of least squares approximations.}
	\end{itemize}

\item
	\ressubheading{The Brower Fixed Point Theorem}{Summer 2000}{Buffalo, NY}{}
	\begin{itemize}
		\resitem{Presented an elementary proof of the Brower Fixed Point Theorem.}
	\end{itemize}
\end{itemize}
